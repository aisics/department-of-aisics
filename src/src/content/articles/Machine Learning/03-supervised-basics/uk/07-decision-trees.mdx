#### –°—Ü–µ–Ω–∞—Ä—ñ–π 3: 80% –í–µ–ª–∏–∫—ñ, 20% –ú–∞–ª—ñ (Gini = 0.32)

$$

2. **Post-pruning (–ü—ñ—Å–ª—è–æ–±—Ä—ñ–∑–∞–Ω–Ω—è):** –í–∏—Ä–æ—â—É—î–º–æ –ø–æ–≤–Ω–µ –¥–µ—Ä–µ–≤–æ, –ø–æ—Ç—ñ–º –≤—ñ–¥—Ä—ñ–∑–∞—î–º–æ –≥—ñ–ª–∫–∏
   - –í–∏—Ä–æ—â—É—î–º–æ –¥–µ—Ä–µ–≤–æ –¥–æ –∫—ñ–Ω—Ü—è
   - –í–∏–¥–∞–ª—è—î–º–æ –≥—ñ–ª–∫–∏, —è–∫—ñ –Ω–µ –ø–æ–ª—ñ–ø—à—É—é—Ç—å –≤–∞–ª—ñ–¥–∞—Ü—ñ–π–Ω—É —Ç–æ—á–Ω—ñ—Å—Ç—å

   - üí° –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–π—Ç–µ `ccp_alpha` (cost complexity pruning) —è–∫ –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤—É

–î–∞–≤–∞–π—Ç–µ –æ–±'—î–¥–Ω–∞—î–º–æ –≤—Å–µ —Ä–∞–∑–æ–º - –ø–æ–±—É–¥—É—î–º–æ, –æ—Ü—ñ–Ω–∏–º–æ —Ç–∞ –ø—Ä–æ–∞–Ω–∞–ª—ñ–∑—É—î–º–æ –¥–µ—Ä–µ–≤–æ:

```python
import pandas as pd
import numpy as np
from sklearn.tree import DecisionTreeClassifier, plot_tree, export_text
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.metrics import (
    accuracy_score, precision_score, recall_score, f1_score,
    confusion_matrix, classification_report, roc_auc_score, roc_curve
)
import matplotlib.pyplot as plt
import seaborn as sns

# –í—Å—Ç–∞–Ω–æ–≤–ª—é—î–º–æ —Å—Ç–∏–ª—å –¥–ª—è –≥—Ä–∞—Ñ—ñ–∫—ñ–≤
sns.set_style("whitegrid")
np.random.seed(42)

# === 1. –ü–Ü–î–ì–û–¢–û–í–ö–ê –î–ê–ù–ò–• ===
print("=" * 50)
print("üìä –ö–†–û–ö 1: –ü–Ü–î–ì–û–¢–û–í–ö–ê –î–ê–ù–ò–•")
print("=" * 50)

# –°—Ç–≤–æ—Ä—é—î–º–æ —Ä–µ–∞–ª—ñ—Å—Ç–∏—á–Ω—ñ –¥–∞–Ω—ñ (–º–æ–∂–Ω–∞ –∑–∞–≤–∞–Ω—Ç–∞–∂–∏—Ç–∏ –≤–∞—à—ñ —Ä–µ–∞–ª—å–Ω—ñ –¥–∞–Ω—ñ)
n_samples = 500
data = pd.DataFrame({
    'temperature': np.random.uniform(0, 35, n_samples),
    'is_friday_evening': np.random.binomial(1, 0.15, n_samples),
    'is_peak_hour': np.random.binomial(1, 0.3, n_samples),
    'is_weekend': np.random.binomial(1, 0.28, n_samples),
    'rolling_avg_3h': np.random.uniform(15, 35, n_samples),
    'price_per_pizza': np.random.uniform(10, 15, n_samples)
})

# –ì–µ–Ω–µ—Ä—É—î–º–æ realistic target
data['order_size'] = (
    (data['is_friday_evening'] * 15) +
    (data['is_peak_hour'] * 8) +
    (-0.3 * data['temperature']) +
    (0.5 * data['rolling_avg_3h']) +
    (data['is_weekend'] * 5) +
    np.random.normal(0, 5, n_samples)
) >= 30
data['order_size'] = data['order_size'].astype(int)

print(f"‚úÖ –ù–∞–±—ñ—Ä –¥–∞–Ω–∏—Ö: {len(data)} –∑–∞–º–æ–≤–ª–µ–Ω—å")
print(f"   –í–µ–ª–∏–∫—ñ: {data['order_size'].sum()} ({data['order_size'].mean()*100:.1f}%)")
print(f"   –ú–∞–ª—ñ: {len(data) - data['order_size'].sum()}")

# === 2. –ù–ê–í–ß–ê–ù–ù–Ø –ú–û–î–ï–õ–Ü ===
print("\n" + "=" * 50)
print("üå≤ –ö–†–û–ö 2: –ù–ê–í–ß–ê–ù–ù–Ø –î–ï–†–ï–í–ê –†–Ü–®–ï–ù–¨")
print("=" * 50)

features = ['temperature', 'is_friday_evening', 'is_peak_hour',
            'is_weekend', 'rolling_avg_3h', 'price_per_pizza']
X = data[features]
y = data['order_size']

X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.2, stratify=y, random_state=42
)

tree = DecisionTreeClassifier(
    max_depth=4,
    min_samples_split=30,
    min_samples_leaf=15,
    random_state=42
)

tree.fit(X_train, y_train)
print(f"‚úÖ –ú–æ–¥–µ–ª—å –Ω–∞–≤—á–µ–Ω–∞!")
print(f"   –ì–ª–∏–±–∏–Ω–∞: {tree.get_depth()}")
print(f"   –õ–∏—Å—Ç–∫–∏: {tree.get_n_leaves()}")

# === 3. –û–¶–Ü–ù–ö–ê –ú–û–î–ï–õ–Ü (–í–ò–ö–û–†–ò–°–¢–û–í–£–Ñ–ú–û ARTICLE 6!) ===
print("\n" + "=" * 50)
print("üìä –ö–†–û–ö 3: –û–¶–Ü–ù–ö–ê –ú–û–î–ï–õ–Ü")
print("=" * 50)

y_pred = tree.predict(X_test)
y_prob = tree.predict_proba(X_test)[:, 1]

print("–ë–∞–∑–æ–≤—ñ –º–µ—Ç—Ä–∏–∫–∏:")
print(f"  Accuracy:  {accuracy_score(y_test, y_pred):.3f}")
print(f"  Precision: {precision_score(y_test, y_pred):.3f}")
print(f"  Recall:    {recall_score(y_test, y_pred):.3f}")
print(f"  F1-Score:  {f1_score(y_test, y_pred):.3f}")
print(f"  ROC AUC:   {roc_auc_score(y_test, y_prob):.3f}")

# –î–µ—Ç–∞–ª—å–Ω–∏–π –∑–≤—ñ—Ç

print(classification_report(y_test, y_pred, target_names=['–ú–∞–ª–µ', '–í–µ–ª–∏–∫–µ']))

# –ú–∞—Ç—Ä–∏—Ü—è –ø–ª—É—Ç–∞–Ω–∏–Ω–∏
cm = confusion_matrix(y_test, y_pred)
print("üî¢ –ú–∞—Ç—Ä–∏—Ü—è –ø–ª—É—Ç–∞–Ω–∏–Ω–∏:")
print(f"              –ü—Ä–æ–≥–Ω–æ–∑")
print(f"            –ú–∞–ª–µ  –í–µ–ª–∏–∫–µ")
print(f"–§–∞–∫—Ç –ú–∞–ª–µ    {cm[0,0]:3d}    {cm[0,1]:3d}")
print(f"     –í–µ–ª–∏–∫–µ  {cm[1,0]:3d}    {cm[1,1]:3d}")

# –ë—ñ–∑–Ω–µ—Å-–≤–ø–ª–∏–≤
print("\nüíº –ë—ñ–∑–Ω–µ—Å-–≤–ø–ª–∏–≤:")
print(f"False Positives: {cm[0,1]} ‚Üí –í—Ç—Ä–∞—Ç–∞ ~{cm[0,1] * 200} –≥—Ä–Ω")
print(f"False Negatives: {cm[1,0]} ‚Üí –í—Ç—Ä–∞—Ç–∞ ~{cm[1,0] * 500} –≥—Ä–Ω")
print(f"–ó–∞–≥–∞–ª—å–Ω—ñ –≤—Ç—Ä–∞—Ç–∏: ~{cm[0,1] * 200 + cm[1,0] * 500} –≥—Ä–Ω")

# === 4. –ö–†–û–°-–í–ê–õ–Ü–î–ê–¶–Ü–Ø ===
print("\n" + "=" * 50)
print("üîÑ –ö–†–û–ö 4: –ö–†–û–°-–í–ê–õ–Ü–î–ê–¶–Ü–Ø")
print("=" * 50)

cv_scores = cross_val_score(tree, X, y, cv=5, scoring='f1')
print(f"F1-Score (5-fold CV): {cv_scores.mean():.3f} (+/- {cv_scores.std() * 2:.3f})")

# === 5. –í–ê–ñ–õ–ò–í–Ü–°–¢–¨ –û–ó–ù–ê–ö ===
print("\n" + "=" * 50)
print("üìä –ö–†–û–ö 5: –í–ê–ñ–õ–ò–í–Ü–°–¢–¨ –û–ó–ù–ê–ö")
print("=" * 50)

importances = pd.DataFrame({
    'feature': features,
    'importance': tree.feature_importances_
}).sort_values('importance', ascending=False)

print(importances.to_string(index=False))

# === 6. –ü–û–Ø–°–ù–ï–ù–ù–Ø –ü–†–û–ì–ù–û–ó–£ ===
print("\n" + "=" * 50)
print("üí° –ö–†–û–ö 6: –ü–û–Ø–°–ù–ï–ù–ù–Ø –ö–û–ù–ö–†–ï–¢–ù–û–ì–û –ü–†–û–ì–ù–û–ó–£")
print("=" * 50)

sample = X_test.iloc[[0]]
prediction = tree.predict(sample)[0]
probability = tree.predict_proba(sample)[0]

print("–í—Ö—ñ–¥–Ω—ñ –¥–∞–Ω—ñ:")
print(sample.to_string())
print(f"\n–ü—Ä–æ–≥–Ω–æ–∑: {'–í–ï–õ–ò–ö–ï' if prediction == 1 else '–ú–ê–õ–ï'}")
print(f"–ô–º–æ–≤—ñ—Ä–Ω–æ—Å—Ç—ñ: –ú–∞–ª–µ {probability[0]:.1%}, –í–µ–ª–∏–∫–µ {probability[1]:.1%}")

# –ü–æ–∫–∞–∑—É—î–º–æ —à–ª—è—Ö —á–µ—Ä–µ–∑ –¥–µ—Ä–µ–≤–æ
print("\n–®–ª—è—Ö —á–µ—Ä–µ–∑ –¥–µ—Ä–µ–≤–æ:")
decision_path = tree.decision_path(sample).toarray()[0]
feature = tree.tree_.feature
threshold = tree.tree_.threshold

for node_id in decision_path.nonzero()[0]:
    if tree.tree_.feature[node_id] != -2:  # Not a leaf
        feat_idx = feature[node_id]
        feat_name = features[feat_idx]
        feat_value = sample.iloc[0, feat_idx]
        thresh = threshold[node_id]

        if feat_value <= thresh:
            print(f"  ‚úì {feat_name} = {feat_value:.2f} <= {thresh:.2f}")
        else:
            print(f"  ‚úó {feat_name} = {feat_value:.2f} > {thresh:.2f}")

print("\n‚úÖ –ê–ù–ê–õ–Ü–ó –ó–ê–í–ï–†–®–ï–ù–û!")
```
